             SELECT

                 fa.status AS facilityStatus, 
                 fa.createdDate AS createdDate
             FROM 
                 Facility fa

                 order by fa.createdDate DESC 
         """ , countQuery = """
             SELECT

                 AND (:#{#request.facilityStatus} IS NULL OR fa.status = :#{#request.facilityStatus})
             
             """
     )
